# --- PDF GENEREERIMINE ---
class PDF(FPDF):
    def header(self):
        self.set_font("Helvetica", 'B', 15)
        self.cell(0, 12, "RISKIANALÜÜS: HALJASTUS JA PUUHOOLDUS", ln=True, align='C')
        self.ln(5)

def create_pdf():
    pdf = PDF()
    pdf.add_page()
    pdf.set_font("Helvetica", size=10)
    
    # Tabeli stiilis andmed
    def draw_row(pdf, label, value):
        pdf.set_font("Helvetica", 'B', 10)
        pdf.cell(50, 8, label, 1)
        pdf.set_font("Helvetica", '', 10)
        pdf.cell(140, 8, str(value), 1, 1)

    draw_row(pdf, "Ettevõte", ettevote)
    draw_row(pdf, "Objekt", aadress)
    draw_row(pdf, "Töö liik", too_liik)
    draw_row(pdf, "Vastutav", vastutav)
    draw_row(pdf, "Kuupäev", kuupaev)
    pdf.ln(5)

    pdf.set_fill_color(230, 240, 230)
    pdf.set_font("Helvetica", 'B', 11)
    pdf.cell(0, 10, " RISKIANALÜÜSI TULEMUSED", 1, 1, 'L', True)
    
    pdf.set_font("Helvetica", size=10)
    riskid = [
        ("Ohuala", ohuala), ("Ilm", ilm), ("Pinnas/Trassid", pinnas),
        ("Vara", vara), ("Liiklus", liiklus), ("Tehnika", tehnika),
        ("Kõrgustööd", korgustood), ("Tervis", ergonoomika)
    ]
    for r_label, r_value in riskid:
        pdf.cell(60, 8, r_label, 1)
        pdf.cell(130, 8, r_value, 1, 1)

    pdf.ln(10)
    pdf.cell(0, 10, f"Kinnitaja: {vastutav} __________________________", ln=True)
    
    return pdf.output(dest='S').encode('latin-1')

if st.button("Genereeri PDF"):
    pdf_bytes = create_pdf()
    st.download_button(label="📥 Laadi PDF alla", data=pdf_bytes, file_name="Riskianalyys.pdf", mime="application/pdf")
